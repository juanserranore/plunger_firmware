1// This header file contains a list of user setup files and defines which one the

9// needed. It also improves the portability of users sketches to other hardware

13// Re-compile and upload after making and saving any changes to this file.

284// These are the pins for ESP8266 boards
285// Name GPIO NodeMCU Function
286#define PIN_D0 16 // GPIO16 WAKE
287#define PIN_D1 5 // GPIO5 User purpose
288#define PIN_D2 4 // GPIO4 User purpose
289#define PIN_D3 0 // GPIO0 Low on boot means enter FLASH mode
290#define PIN_D4 2 // GPIO2 TXD1 (must be high on boot to go to UART0 FLASH mode)
291#define PIN_D5 14 // GPIO14 HSCLK
292#define PIN_D6 12 // GPIO12 HMISO
293#define PIN_D7 13 // GPIO13 HMOSI RXD2
294#define PIN_D8 15 // GPIO15 HCS TXD0 (must be low on boot to enter UART0 FLASH mode)
295#define PIN_D9 3 // RXD0
296#define PIN_D10 1 // TXD0
297
298#define PIN_MOSI 8 // SD1 FLASH and overlap mode
299#define PIN_MISO 7 // SD0
300#define PIN_SCLK 6 // CLK
301#define PIN_HWCS 0 // D3
302
303#define PIN_D11 9 // SD2
304#define PIN_D12 10 // SD4
